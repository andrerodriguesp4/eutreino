const exerciciosAdOrdenado = exerciciosAd.sort();

async function adicionarExerciciosNoFirestore() {
  try {
    const exerciciosCollection = collection(db, 'exercicios');

    for (let i = 0; i < exerciciosAdOrdenado.length; i++) {
      const titulo = exerciciosAdOrdenado[i];
      const id = i.toString().padStart(3, '0'); // Garante IDs como '001', '002', etc.

      await setDoc(doc(exerciciosCollection, id), {
        id,
        titulo,
      });
    }

    console.log('Exercícios adicionados em ordem alfabética com IDs formatados.');
  } catch (error) {
    console.error('Erro ao adicionar exercícios:', error);
  }
}
